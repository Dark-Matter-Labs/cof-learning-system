```typescript
// src/components/shared/CaptureModalContext.tsx
'use client';

import { createContext, useContext, useState, type ReactNode } from 'react';

interface CaptureModalContextValue {
  readonly isOpen: boolean;
  readonly openCapture: () => void;
  readonly closeCapture: () => void;
}

const CaptureModalContext = createContext<CaptureModalContextValue | null>(null);

export function CaptureModalProvider({ children }: { readonly children: ReactNode }) {
  const [isOpen, setIsOpen] = useState(false);
  const openCapture = () => setIsOpen(true);
  const closeCapture = () => setIsOpen(false);
  return (
    <CaptureModalContext.Provider value={{ isOpen, openCapture, closeCapture }}>
      {children}
    </CaptureModalContext.Provider>
  );
}

export function useCaptureModal(): CaptureModalContextValue {
  const ctx = useContext(CaptureModalContext);
  if (!ctx) throw new Error('useCaptureModal must be used inside CaptureModalProvider');
  return ctx;
}
```

```typescript
// src/components/shared/CaptureModal.tsx
'use client';

import { useState } from 'react';
import { useCaptureModal } from './CaptureModalContext';
import { QuickCaptureForm, type CaptureFormData, type EntryMode } from '@/components/capture/QuickCaptureForm';

type SubmitState = 'idle' | 'submitting' | 'done';

export function CaptureModal() {
  const { isOpen, closeCapture } = useCaptureModal();
  const [submitState, setSubmitState] = useState<SubmitState>('idle');
  const [entryMode, setEntryMode] = useState<EntryMode>('call');
  const [summary, setSummary] = useState<string | null>(null);

  const handleSubmit = async (data: CaptureFormData) => {
    setSubmitState('submitting');
    try {
      const node_type = entryMode === 'call' ? 'meeting_notes' : 'hunch';
      const res = await fetch('/api/capture', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          title: data.title,
          description: data.description,
          node_type,
          insight_date: data.date,
          participant_ids: data.participant_ids,
          attachment: data.attachment,
        }),
      });
      if (!res.ok) throw new Error('Capture failed');
      setSummary('submitted');
      setSubmitState('done');
    } catch {
      setSubmitState('idle');
    }
  };

  const handleCaptureAnother = () => {
    setSubmitState('idle');
    setSummary(null);
  };

  if (!isOpen) return null;

  return (
    <div className="fixed inset-0 z-[100] flex items-center justify-center p-4">
      <div className="absolute inset-0 bg-black/40 backdrop-blur-sm" onClick={closeCapture} />
      <div className="relative w-full max-w-lg bg-white dark:bg-gray-950 rounded-2xl shadow-2xl border border-gray-200/80 dark:border-gray-800/50 overflow-hidden">

        <div className="flex items-center justify-between px-6 py-4 border-b border-gray-100 dark:border-gray-800">
          <h2 className="text-sm font-medium text-gray-900 dark:text-gray-100">Capture</h2>
          <button onClick={closeCapture} title="Close" className="w-7 h-7 flex items-center justify-center text-gray-400 hover:text-gray-600 dark:hover:text-gray-300 rounded-lg hover:bg-gray-100 dark:hover:bg-gray-800 text-lg leading-none">×</button>
        </div>

        <div className="p-6">
          {submitState === 'done' && summary ? (
            <div className="space-y-4">
              <p className="text-sm text-gray-500">Your capture was submitted and is being processed.</p>
              <div className="flex gap-3">
                <button onClick={handleCaptureAnother} className="text-sm text-gray-500 hover:text-gray-700 underline">Capture another</button>
                <button onClick={closeCapture} className="px-4 py-2 bg-gray-900 dark:bg-gray-100 text-white dark:text-gray-900 rounded-lg text-sm font-medium">Done</button>
              </div>
            </div>
          ) : (
            <div className="space-y-5">
              <div className="flex gap-2">
                {(['call', 'thought', 'file'] as EntryMode[]).map(mode => (
                  <button
                    key={mode}
                    onClick={() => setEntryMode(mode)}
                    className={`px-3 py-1.5 rounded-lg text-xs font-medium transition-colors ${
                      entryMode === mode
                        ? 'bg-gray-900 dark:bg-gray-100 text-white dark:text-gray-900'
                        : 'text-gray-400 hover:text-gray-600 hover:bg-gray-100 dark:hover:bg-gray-800'
                    }`}
                  >
                    {mode === 'call' ? 'Meeting notes' : mode === 'thought' ? 'Quick thought' : 'Upload file'}
                  </button>
                ))}
              </div>

              <QuickCaptureForm
                onSubmit={handleSubmit}
                isSubmitting={submitState === 'submitting'}
                entryMode={entryMode}
              />
            </div>
          )}
        </div>
      </div>
    </div>
  );
}
```

```typescript
'use client';

import { useEffect } from 'react';
import { useCaptureModal } from './CaptureModalContext';

export function GlobalKeyboardHandler() {
  const { openCapture } = useCaptureModal();

  useEffect(() => {
    const handler = (e: KeyboardEvent) => {
      if ((e.metaKey || e.ctrlKey) && e.key === 'n') {
        e.preventDefault();
        openCapture();
      }
    };
    window.addEventListener('keydown', handler);
    return () => window.removeEventListener('keydown', handler);
  }, [openCapture]);

  return null;
}
```

```typescript
// src/app/api/log/route.ts
import { createClient } from '@/lib/supabase/server';
import { NextResponse } from 'next/server';

const LOG_NODE_TYPES = [
  'hunch', 'learning', 'signal', 'commitment', 'option', 'test',
  'assumption_background', 'assumption_foreground', 'meeting_notes',
];

export async function GET() {
  const supabase = await createClient();
  const { data: { user }, error: authError } = await supabase.auth.getUser();
  if (authError || !user) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });

  const { data, error } = await supabase
    .from('nodes')
    .select('id, node_type, title, description, status, content, created_at, author_id')
    .eq('status', 'promoted')
    .neq('node_type', 'person')
    .order('created_at', { ascending: false })
    .limit(100);

  if (error) return NextResponse.json({ error: error.message }, { status: 500 });

  const filtered = (data ?? []).filter((n: { node_type: string }) => LOG_NODE_TYPES.includes(n.node_type));
  return NextResponse.json({ data: filtered }, { status: 200 });
}
```

```typescript
// src/components/log/ActivityCard.tsx
import Link from 'next/link';

interface LogNode {
  readonly id: string;
  readonly node_type: string;
  readonly title: string;
  readonly description: string | null;
  readonly status: string;
  readonly content: Record<string, unknown> | null;
  readonly created_at: string;
  readonly author_id: string;
}

interface Props {
  readonly node: LogNode;
}

function getBadgeLabel(node: LogNode): string {
  if (node.content?.is_decision) return 'Decision';
  if (node.content?.is_digest) return 'Weekly digest';
  switch (node.node_type) {
    case 'meeting_notes': return 'Meeting';
    case 'signal': return 'Signal';
    case 'learning': return 'Learning';
    case 'commitment': return 'Commitment';
    case 'option': return 'Option';
    case 'test': return 'Test';
    default: return 'Insight';
  }
}

function getBadgeColor(label: string): string {
  switch (label) {
    case 'Meeting': return 'bg-blue-50 text-blue-700 dark:bg-blue-900/30 dark:text-blue-300';
    case 'Decision': return 'bg-purple-50 text-purple-700 dark:bg-purple-900/30 dark:text-purple-300';
    case 'Signal': return 'bg-yellow-50 text-yellow-700 dark:bg-yellow-900/30 dark:text-yellow-300';
    case 'Weekly digest': return 'bg-green-50 text-green-700 dark:bg-green-900/30 dark:text-green-300';
    default: return 'bg-gray-50 text-gray-600 dark:bg-gray-800 dark:text-gray-300';
  }
}

function formatTime(iso: string): string {
  return new Date(iso).toLocaleTimeString([], { hour: 'numeric', minute: '2-digit' });
}

export function ActivityCard({ node }: Props) {
  const badge = getBadgeLabel(node);
  const badgeColor = getBadgeColor(badge);

  return (
    <div className="p-4 border border-gray-100 dark:border-gray-800 rounded-xl hover:border-gray-200 dark:hover:border-gray-700 transition-colors">
      <div className="flex items-start justify-between gap-3">
        <div className="flex-1 min-w-0">
          <div className="flex items-center gap-2 mb-1">
            <span className={`text-xs px-2 py-0.5 rounded-full font-medium ${badgeColor}`}>{badge}</span>
            <span className="text-xs text-gray-400">{formatTime(node.created_at)}</span>
          </div>
          <p className="text-sm font-medium text-gray-900 dark:text-gray-100 truncate">{node.title}</p>
          {node.description && (
            <p className="text-xs text-gray-500 mt-0.5 line-clamp-2">{node.description}</p>
          )}
        </div>
        <Link
          href={`/?node=${node.id}`}
          className="text-xs text-gray-400 hover:text-gray-600 whitespace-nowrap mt-1 shrink-0"
        >
          View →
        </Link>
      </div>
    </div>
  );
}
```

```typescript
// src/app/log/LogClient.tsx
'use client';

import { useState, useMemo } from 'react';
import { ActivityCard } from '@/components/log/ActivityCard';
import { LogFilters, type LogFilter } from '@/components/log/LogFilters';
import { useCaptureModal } from '@/components/shared/CaptureModalContext';

interface LogNode {
  readonly id: string;
  readonly node_type: string;
  readonly title: string;
  readonly description: string | null;
  readonly status: string;
  readonly content: Record<string, unknown> | null;
  readonly created_at: string;
  readonly author_id: string;
}

interface Props {
  readonly nodes: ReadonlyArray<LogNode>;
}

function groupByDate(nodes: ReadonlyArray<LogNode>): Map<string, LogNode[]> {
  const groups = new Map<string, LogNode[]>();
  const today = new Date().toDateString();
  const yesterday = new Date(Date.now() - 86400000).toDateString();

  for (const node of nodes) {
    const date = new Date(node.created_at).toDateString();
    let label: string;
    if (date === today) label = 'Today';
    else if (date === yesterday) label = 'Yesterday';
    else {
      label = new Date(node.created_at).toLocaleDateString([], { weekday: 'long', month: 'short', day: 'numeric' });
    }
    const group = groups.get(label) ?? [];
    groups.set(label, [...group, node]);
  }
  return groups;
}

function matchesFilter(node: LogNode, filter: LogFilter): boolean {
  if (filter === 'all') return true;
  if (filter === 'decision') return Boolean(node.content?.is_decision);
  return node.node_type === filter;
}

export function LogClient({ nodes }: Props) {
  const [filter, setFilter] = useState<LogFilter>('all');
  const { openCapture } = useCaptureModal();

  const filtered = useMemo(() => nodes.filter(n => matchesFilter(n, filter)), [nodes, filter]);
  const grouped = useMemo(() => groupByDate(filtered), [filtered]);

  return (
    <div className="max-w-2xl mx-auto px-4 py-8 space-y-8">
      <div className="flex items-center justify-between">
        <h1 className="text-lg font-medium text-gray-900 dark:text-gray-100">Activity log</h1>
        <button
          onClick={openCapture}
          className="text-xs px-3 py-1.5 bg-gray-900 dark:bg-gray-100 text-white dark:text-gray-900 rounded-lg font-medium hover:opacity-90"
        >
          + Capture
        </button>
      </div>

      <LogFilters active={filter} onChange={setFilter} />

      {grouped.size === 0 && (
        <div className="text-center py-16 text-sm text-gray-400">No activity yet. Start by capturing something.</div>
      )}

      {Array.from(grouped.entries()).map(([dateLabel, dateNodes]) => (
        <div key={dateLabel} className="space-y-3">
          <p className="text-xs font-medium text-gray-400 uppercase tracking-widest">{dateLabel}</p>
          <div className="space-y-2">
            {dateNodes.map(node => <ActivityCard key={node.id} node={node} />)}
          </div>
        </div>
      ))}
    </div>
  );
}
```

```typescript
// src/app/api/attention/route.ts
import { createClient } from '@/lib/supabase/server';
import { NextResponse } from 'next/server';

interface AttentionItem {
  readonly type: 'orphan' | 'stale_commitment' | 'unresolved_tension' | 'pending_review';
  readonly title: string;
  readonly description: string;
  readonly href: string;
  readonly severity: 'high' | 'medium' | 'low';
}

const STALE_DAYS = 10;
const ORPHAN_DAYS = 14;

function daysAgo(days: number): string {
  return new Date(Date.now() - days * 86400000).toISOString();
}

export async function GET() {
  const supabase = await createClient();
  const { data: { user }, error: authError } = await supabase.auth.getUser();
  if (authError || !user) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });

  const items: AttentionItem[] = [];

  // 1. Pending review items
  const { count: reviewCount } = await supabase
    .from('nodes')
    .select('*', { count: 'exact', head: true })
    .eq('status', 'flagged_for_review');
  if ((reviewCount ?? 0) > 0) {
    items.push({
      type: 'pending_review',
      title: `${reviewCount} items need review`,
      description: 'Items flagged by the system for human review.',
      href: '/review',
      severity: 'high',
    });
  }

  // 2. Orphan captures (promoted nodes created in last 14 days with no edges)
  const { data: recentNodes } = await supabase
    .from('nodes')
    .select('id, title')
    .eq('status', 'promoted')
    .neq('node_type', 'person')
    .neq('node_type', 'goal_space')
    .gt('created_at', daysAgo(ORPHAN_DAYS))
    .limit(50);

  if (recentNodes && recentNodes.length > 0) {
    let orphanCount = 0;
    for (const node of recentNodes) {
      const { count } = await supabase
        .from('edges')
        .select('*', { count: 'exact', head: true })
        .eq('source_id', node.id)
        .limit(1);
      if ((count ?? 0) === 0) orphanCount++;
    }
    if (orphanCount > 0) {
      items.push({
        type: 'orphan',
        title: `${orphanCount} unconnected captures`,
        description: `${orphanCount} recent items haven't been linked to any goal or assumption.`,
        href: '/review',
        severity: 'medium',
      });
    }
  }

  // 3. Stale commitments (commitment nodes with no edge activity in STALE_DAYS)
  const { data: commitments } = await supabase
    .from('nodes')
    .select('id, title')
    .eq('node_type', 'commitment')
    .eq('status', 'promoted')
    .lt('updated_at', daysAgo(STALE_DAYS))
    .limit(5);

  for (const c of commitments ?? []) {
    items.push({
      type: 'stale_commitment',
      title: c.title,
      description: `No activity for ${STALE_DAYS}+ days.`,
      href: '/commitments',
      severity: 'medium',
    });
  }

  // Sort: high severity first
  const sorted = [...items].sort((a, b) => {
    const order = { high: 0, medium: 1, low: 2 };
    return order[a.severity] - order[b.severity];
  });

  return NextResponse.json({ data: sorted.slice(0, 4) }, { status: 200 });
}
```

```typescript
// src/components/shared/AttentionCard.tsx
'use client';

import { useEffect, useState } from 'react';
import Link from 'next/link';

interface AttentionItem {
  readonly type: string;
  readonly title: string;
  readonly description: string;
  readonly href: string;
  readonly severity: 'high' | 'medium' | 'low';
}

const DISMISSED_KEY = 'attention_dismissed_at';
const DISMISS_DURATION_MS = 8 * 60 * 60 * 1000; // 8 hours

function isDismissed(): boolean {
  const ts = sessionStorage.getItem(DISMISSED_KEY);
  if (!ts) return false;
  return Date.now() - parseInt(ts, 10) < DISMISS_DURATION_MS;
}

function severityIcon(severity: 'high' | 'medium' | 'low'): string {
  if (severity === 'high') return '🔴';
  if (severity === 'medium') return '⚠';
  return '📌';
}

export function AttentionCard() {
  const [items, setItems] = useState<AttentionItem[] | null>(null);
  const [dismissed, setDismissed] = useState(false);

  useEffect(() => {
    if (isDismissed()) { setDismissed(true); return; }
    fetch('/api/attention')
      .then(r => r.json())
      .then(({ data }) => setItems(data ?? []))
      .catch(() => setItems([]));
  }, []);

  const dismiss = () => {
    sessionStorage.setItem(DISMISSED_KEY, String(Date.now()));
    setDismissed(true);
  };

  if (dismissed || items === null || items.length === 0) return null;

  return (
    <div className="border border-gray-200 dark:border-gray-700 rounded-xl p-4 space-y-3">
      <div className="flex items-center justify-between">
        <p className="text-xs font-medium text-gray-700 dark:text-gray-300 uppercase tracking-wide">What needs attention</p>
        <button onClick={dismiss} title="Dismiss" className="text-gray-400 hover:text-gray-600 text-lg leading-none">×</button>
      </div>
      <div className="space-y-2">
        {items.map((item, i) => (
          <div key={i} className="flex items-start justify-between gap-3">
            <div className="flex gap-2 items-start">
              <span className="text-sm mt-0.5">{severityIcon(item.severity)}</span>
              <div>
                <p className="text-sm text-gray-700 dark:text-gray-300">{item.title}</p>
                {item.description && <p className="text-xs text-gray-400 mt-0.5">{item.description}</p>}
              </div>
            </div>
            <Link href={item.href} className="text-xs text-gray-400 hover:text-gray-600 whitespace-nowrap shrink-0">Review →</Link>
          </div>
        ))}
      </div>
    </div>
  );
}
```
